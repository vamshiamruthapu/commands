- hosts: dev
  become: true
  vars:
    jdk_download_url: https://download.java.net/java/GA/jdk18.0.2.1/db379da656dc47308e138f21b33976fa/1/GPL/openjdk-18.0.2.1_linux-x64_bin.tar.gz
    jdk_download_location: /opt/openjdk-18.0.2.1_linux-x64_bin.tar.gz
    java_home: /opt/jdk-18.0.2.1
  tasks:
    - name: Download the JDK binaries
      get_url:
        url: "{{ jdk_download_url }}"
        dest: "{{ jdk_download_location }}"
    - name: Unzip the downloaded file
      unarchive:
        src: "{{ jdk_download_location }}"
        dest: /opt
        remote_src: yes
    - name: Set the JAVA_HOME in /etc/profile file
      lineinfile:
        path: /etc/profile.d/java.sh
        create: yes
        state: present
        line: "{{ item }}"
      with_items:
        - 'export JAVA_HOME="{{ java_home }}"'
        - 'export PATH=$PATH:$JAVA_HOME/bin'
    - name: Reload /etc/profile.d/java.sh file
      shell:
        cmd: source /etc/profile.d/java.sh
